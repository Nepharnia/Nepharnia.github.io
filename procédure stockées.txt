Récupération de IdUsr :

DELIMITER | 
CREATE PROCEDURE recup_idUsr (IN mon_utilisateur CHAR(50), OUT idUsr_recup INT)
BEGIN
SELECT idUsr INTO idUsr_recup
FROM authentification, avoir, user, infos 
WHERE user.loginUsr = mon_utilisateur AND user.idUsr = infos.idUsrfk AND infos.idUsrfk = avoir.idUsr_fk AND avoir.idA_fk = authentification.idA;
END |
DELIMITER ;
